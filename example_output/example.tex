\section{Discovered Circuit: Fourier Multiplication Algorithm}

The model computes $(a + b) \mod 113$ using 3 key frequencies $k \in \{3, 38, 51\}$ with angular frequencies $\omega_k = \frac{2\pi k}{113}$.

\subsection{Step 1: Embedding (Token $\to$ Fourier Components)}

The embedding matrix $W_E$ maps each one-hot token $t \in \{0, \ldots, 112\}$ onto sinusoidal components:

\begin{align}
    \mathbf{x}^{(0)}_a &= \underbrace{W_E \cdot \mathbf{e}_a}_{\text{token embedding of } a} + \underbrace{\mathbf{p}_0}_{\text{positional embedding (pos 0)}} \\
    &\approx \sum_{k \in \mathcal{K}} \bigg[ \underbrace{\alpha_k \cos\!\left(\frac{2\pi k \cdot a}{113}\right)}_{\substack{\text{cosine component} \\ \text{from } W_E \text{ row } a}} \cdot \mathbf{u}_k^{(\cos)} + \underbrace{\beta_k \sin\!\left(\frac{2\pi k \cdot a}{113}\right)}_{\substack{\text{sine component} \\ \text{from } W_E \text{ row } a}} \cdot \mathbf{u}_k^{(\sin)} \bigg]
\end{align}

where $\mathcal{K} = \{3, 38, 51\}$ are the key frequencies discovered via Fourier analysis of $W_E$.

Similarly for token $b$ at position 1:
\begin{align}
    \mathbf{x}^{(0)}_b &= \underbrace{W_E \cdot \mathbf{e}_b}_{\text{token embedding of } b} + \underbrace{\mathbf{p}_1}_{\text{positional embedding (pos 1)}}
\end{align}

\subsection{Step 2: Attention (Move Fourier Info to Output Position)}

The attention heads compute scores from the `$=$' token (position 2) to positions 0 and 1, then produce a weighted combination:

\begin{align}
    \text{score}^{(j)}_{=\to a} &= \frac{\overbrace{\mathbf{x}^{(0)}_= \cdot W_Q^{(j)}}^{\text{query from `=' position}}\cdot \overbrace{\left(W_K^{(j)}\right)^\top \!\cdot \mathbf{x}^{(0)}_a}^{\text{key from position } a}}{\underbrace{\sqrt{d_{\text{head}}}}_{\text{scaling factor}}} \\
    A^{(j)}_0 &= \underbrace{\sigma\!\left(\text{score}^{(j)}_{=\to a} - \text{score}^{(j)}_{=\to b}\right)}_{\substack{\text{softmax over 2 elements} \\ \text{= sigmoid of difference}}} \\
    &\approx \underbrace{0.5}_{\text{uniform baseline}} + \underbrace{\gamma_j \Big(\cos\!\left(\omega_{k_j} \cdot a\right) - \cos\!\left(\omega_{k_j} \cdot b\right)\Big)}_{\substack{\text{periodic modulation at frequency } k_j \\\text{(from } C^{(j)} = W_E^\top W_K^{(j)\top} W_Q^{(j)} \mathbf{x}^{(0)}_= \text{)}}} \\
    \text{attn\_out}^{(j)} &= \underbrace{A^{(j)}_0}_{\text{attn weight to } a} \cdot \overbrace{W_O^{(j)} W_V^{(j)} \mathbf{x}^{(0)}_a}^{\text{OV circuit applied to } a} + \underbrace{A^{(j)}_1}_{\text{attn weight to } b} \cdot \overbrace{W_O^{(j)} W_V^{(j)} \mathbf{x}^{(0)}_b}^{\text{OV circuit applied to } b}
\end{align}

\textbf{Key insight:} Since $A^{(j)}_0$ is approximately linear in $\cos(\omega_{k_j} a)$ and the OV circuit outputs $\cos(\omega_{k_j} a)$, their product creates \emph{degree-2 polynomials} of sines and cosines --- exactly what is needed for the trig identity in the next step.

\subsection{Step 3: MLP (Compute Trig Identities)}

The residual stream after attention contains degree-2 products. The MLP neurons compute the key trigonometric identities:

\begin{align}
    \mathbf{x}^{(1)} &= \underbrace{\mathbf{x}^{(0)}_=}_{\substack{\text{skip connection} \\ \text{(= token embedding)}}} + \sum_{j=0}^{3} \underbrace{\text{attn\_out}^{(j)}}_{\substack{\text{attention head } j \\ \text{output (degree-2 trig)}}} \\
    \text{pre}_n &= \underbrace{W_{\text{in}}[n, :] \cdot \mathbf{x}^{(1)}}_{\text{linear projection of neuron } n} + \underbrace{b_{\text{in}}[n]}_{\text{bias}} \\
    \text{MLP}[n] &= \underbrace{\text{ReLU}\!\left(\text{pre}_n\right)}_{\substack{\text{activation of neuron } n \\ \approx \text{ degree-2 polynomial of } \cos(\omega_k a), \sin(\omega_k a), \ldots}} \\
\end{align}

\textbf{The core computation:} For each key frequency $k$, the MLP neurons collectively compute:

\begin{align}
    \underbrace{\mathbf{u}_k^\top \cdot \text{MLP}(a,b)}_{\substack{\text{projection of MLP activations} \\ \text{onto direction } \mathbf{u}_k \text{ in } W_L}} &\approx \underbrace{\alpha_k \cos\!\left(\omega_k(a+b)\right)}_{\text{target: cosine of sum}} \\
    &= \overbrace{\underbrace{\alpha_k \cos(\omega_k a) \cos(\omega_k b)}_{\substack{\text{from neurons computing} \\ \cos(\omega_k a)\cos(\omega_k b)}} - \underbrace{\alpha_k \sin(\omega_k a) \sin(\omega_k b)}_{\substack{\text{from neurons computing} \\ \sin(\omega_k a)\sin(\omega_k b)}}}^{\text{cosine addition formula}} \\
    &\quad \notag \\
    \underbrace{\mathbf{v}_k^\top \cdot \text{MLP}(a,b)}_{\substack{\text{projection onto} \\ \text{direction } \mathbf{v}_k \text{ in } W_L}} &\approx \underbrace{\beta_k \sin\!\left(\omega_k(a+b)\right)}_{\text{target: sine of sum}} \\
    &= \overbrace{\underbrace{\beta_k \sin(\omega_k a) \cos(\omega_k b)}_{\substack{\text{from neurons computing} \\ \sin(\omega_k a)\cos(\omega_k b)}} + \underbrace{\beta_k \cos(\omega_k a) \sin(\omega_k b)}_{\substack{\text{from neurons computing} \\ \cos(\omega_k a)\sin(\omega_k b)}}}^{\text{sine addition formula}}
\end{align}

where $\mathbf{u}_k, \mathbf{v}_k \in \mathbb{R}^{512}$ are the directions in the neuron-logit map $W_L = W_{\text{out}}^\top W_U^\top$ corresponding to $\cos(\omega_k c)$ and $\sin(\omega_k c)$ respectively.

\subsection{Step 4: Unembedding (Fourier $\to$ Logits)}

The neuron-logit map $W_L = W_{\text{out}}^\top \cdot W_U^\top$ converts the MLP activations to output logits:

\begin{align}
    W_L &\approx \sum_{k \in \mathcal{K}} \bigg[ \underbrace{\cos(\omega_k c)}_{\substack{\text{vector in } \mathbb{R}^P \\\text{(c-th entry} = \cos(\omega_k c)\text{)}}} \cdot \underbrace{\mathbf{u}_k^\top}_{\substack{\text{direction in neuron space} \\\text{that reads } \cos(\omega_k(a+b))}} + \underbrace{\sin(\omega_k c)}_{\substack{\text{vector in } \mathbb{R}^P \\\text{(c-th entry} = \sin(\omega_k c)\text{)}}} \cdot \underbrace{\mathbf{v}_k^\top}_{\substack{\text{direction in neuron space} \\\text{that reads } \sin(\omega_k(a+b))}} \bigg] \\
\end{align}

Therefore, the logit for output class $c$ is:

\begin{align}
    \text{Logit}(c \mid a, b) &= \underbrace{W_L \cdot \text{MLP}(a,b)}_{\text{neuron-logit map applied to MLP output}} \\
    &\approx \sum_{k \in \mathcal{K}} \bigg[ \underbrace{\cos(\omega_k c)}_{\substack{\text{from } W_L \\ \text{(unembed direction)}}} \cdot \underbrace{\mathbf{u}_k^\top \text{MLP}(a,b)}_{\substack{\text{MLP computes this} \\ \approx \alpha_k\cos(\omega_k(a+b))}} + \underbrace{\sin(\omega_k c)}_{\substack{\text{from } W_L \\ \text{(unembed direction)}}} \cdot \underbrace{\mathbf{v}_k^\top \text{MLP}(a,b)}_{\substack{\text{MLP computes this} \\ \approx \beta_k\sin(\omega_k(a+b))}} \bigg] \\
    &\approx \sum_{k \in \mathcal{K}} \underbrace{\alpha_k \Big[\cos(\omega_k c)\cos(\omega_k(a+b)) + \sin(\omega_k c)\sin(\omega_k(a+b))\Big]}_{\text{cosine subtraction identity}} \\
    &= \boxed{\sum_{k \in \mathcal{K}} \underbrace{\alpha_k \cos\!\left(\omega_k(a + b - c)\right)}_{\substack{\text{peaks when } c \equiv a+b \pmod{113} \\\text{since } \cos(0) = 1 \text{ is maximum}}}}
\end{align}

\subsection{Step 5: Prediction via Constructive Interference}

\begin{align}
    \hat{c} &= \underbrace{\arg\max_{c \in \{0,\ldots,112\}}}_{\text{select class with highest logit}} \overbrace{\sum_{k \in \mathcal{K}} \alpha_k \cos\!\left(\frac{2\pi k (a + b - c)}{113}\right)}^{\text{sum of cosines at key frequencies}} \\
    &= \underbrace{(a + b) \mod 113}_{\substack{\text{constructive interference:} \\\text{all cosines equal 1 when } c = a+b \bmod 113 \\\text{destructive interference elsewhere}}}
\end{align}

\subsection{Why Multiple Frequencies? (Constructive Interference)}

A single cosine $\cos(\omega_k x)$ has period $113$ but near-maxima at other values of $x$. By summing over multiple key frequencies, the model creates a function with a \emph{unique} maximum at $x = 0 \bmod 113$:

\begin{align}
    f(x) &= \sum_{k \in \mathcal{K}} \underbrace{\cos\!\left(\frac{2\pi k \cdot x}{113}\right)}_{\text{each has max at } x=0} \\
    f(0) &= \underbrace{3}_{\text{all } 3 \text{ cosines} = 1} \quad \gg \quad f(x \neq 0) \quad \underbrace{\text{(destructive interference)}}_{\text{cosines at different frequencies cancel}}
\end{align}

\end{equation*}

For any $c \neq 20$, the cosines at different frequencies point in different directions and \textbf{destructively interfere}, giving a smaller logit.
